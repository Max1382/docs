import { useEffect, useState } from 'react';

export default function TestComponent() {
  const [message, setMessage] = useState('');

  useEffect(() => {
    // Fetch the data from the PHP file
    fetch('./test.php')
      .then((response) => response.text())
      .then((data) => {
        setMessage(data);
      });
  }, []);

  return (
    <>
      <h1>PHP Output in MDX</h1>
      <p>{message}</p>
    </>
  );
}
